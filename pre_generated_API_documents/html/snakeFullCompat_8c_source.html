61pid_t _snake_fork(void) {
62 STARTUPINFO si = { sizeof(si) };
63 PROCESS_INFORMATION pi;
64 char cmdline[MAX_PATH];
65
66 GetModuleFileName(NULL, cmdline, MAX_PATH);
67 if (CreateProcess(NULL, cmdline, NULL, NULL, FALSE,
68 CREATE_NEW_CONSOLE, NULL, NULL, &si, &pi)) {
69 CloseHandle(pi.hThread);
70 return (pid_t)pi.dwProcessId;
71 }
72 return (pid_t)-1;
73}

83pid_t _snake_waitpid(pid_t pid, int *status, int options) {
84 HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, (DWORD)pid);
85 if (hProcess) {
86 WaitForSingleObject(hProcess, INFINITE);
87 GetExitCodeProcess(hProcess, (DWORD*)status);
88 CloseHandle(hProcess);
89 return pid;
90 }
91 return (pid_t)-1;
92}
